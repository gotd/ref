// Type of a Definition or a Parameter.
type Type struct {
	Namespace  []string `json:"namespace,omitempty"`   // namespace components of the type
	Name       string   `json:"name,omitempty"`        // the name of the type
	Bare       bool     `json:"bare,omitempty"`        // whether this type is bare or boxed
	Percent    bool     `json:"-"`                     // whether this type has percent in name (like %Message)
	GenericRef bool     `json:"generic_ref,omitempty"` // whether the type name refers to a generic definition
	GenericArg *Type    `json:"generic_arg,omitempty"` // generic arguments of the type
}

func ( Type) () string {
	var  strings.Builder
	if .Percent {
		.WriteByte('%')
	}
	if .GenericRef {
		.WriteByte('!')
	}
	for ,  := range .Namespace {
		.WriteString()
		.WriteByte('.')
	}
	.WriteString(.Name)
	if .GenericArg != nil {
		.WriteByte('<')
		.WriteString(.GenericArg.())
		.WriteByte('>')
	}
	return .String()
}

func ( *Type) ( string) error {
	if len() < 1 {
		return errors.New("got empty string")
	}

	switch [0] {
	case '.':
		return errors.New("type can't start with dot")
	case '%':
		.Bare = true
		.Percent = true
		 = [1:]
	case '!':
		.GenericRef = true
		 = [1:]
	}

	// Parse `type<generic_arg>`.
	if  := strings.Index(, "<");  >= 0 {
		if !strings.HasSuffix(, ">") {
			return errors.New("invalid generic")
		}
		.GenericArg = &Type{}
		if  := .GenericArg.([+1 : len()-1]);  != nil {
			return fmt.Errorf("failed to parse generic: %w", )
		}
		 = [:]
	}

	// Parse `ns1.ns2.name`.
	 := strings.Split(, ".")
	if len() == 1 {
		.Name = [0]
	} else {
		.Name = [len()-1]
		.Namespace = [:len()-1]
	}
	if .Name == "" {
		return errors.New("blank name")
	}
	if !isValidName(.Name) {
		return fmt.Errorf("invalid name %q", .Name)
	}
	for ,  := range .Namespace {
		if !isValidName() {
			return fmt.Errorf("invalid namespace part %q", )
		}
	}

	// Bare types starts from lowercase.
	if len(.Name) > 0 && !.Percent {
		.Bare = .Name[0:1] == strings.ToLower(.Name[0:1])
	}
	return nil
}
