// Parameter with Name and Type.
type Parameter struct {
	// Name of Parameter.
	Name string `json:"name,omitempty"`
	// Type of Parameter.
	Type Type `json:"type"`
	// Flag specifies flag name and index if parameter is conditional.
	Flag *Flag `json:"flag,omitempty"`
	// Flags denotes whether Parameter is flags field (uint32).
	//
	// If true, Type and Flag are blank.
	Flags bool `json:"flags,omitempty"`

	// special case for {X:Type} definitions aka generic parameters,
	// only "Name" field is populated.
	typeDefinition bool
}

func ( Parameter) () bool {
	return .Flag != nil
}

func ( Parameter) () string {
	var  strings.Builder
	if .typeDefinition {
		.WriteRune('{')
	}
	if .Name != "" {
		// Anonymous parameter.
		.WriteString(.Name)
		.WriteRune(':')
	}
	if .typeDefinition {
		.WriteString("Type}")
		return .String()
	}
	if .Flags {
		.WriteRune('#')
		return .String()
	}
	if .Conditional() {
		.WriteString(.Flag.String())
		.WriteRune('?')
	}
	.WriteString(.Type.String())
	return .String()
}
