// Options of rpc engine.
type Options struct {
	RetryInterval time.Duration
	MaxRetries    int
	Logger        *zap.Logger
	Clock         clock.Clock
	DropHandler   DropHandler
}

func ( *Options) () {
	if .RetryInterval == 0 {
		.RetryInterval = time.Second * 10
	}
	if .MaxRetries == 0 {
		.MaxRetries = 5
	}
	if .Logger == nil {
		.Logger = zap.NewNop()
	}
	if .Clock == nil {
		.Clock = clock.System
	}
	if .DropHandler == nil {
		.DropHandler = NopDrop
	}
}
