// createAuthKey generates new authorization key.
func ( *Conn) ( context.Context) error {
	,  := exchange.NewExchanger(.conn).
		WithClock(.clock).
		WithLogger(.log.Named("exchange")).
		WithTimeout(.exchangeTimeout).
		WithRand(.rand).
		Client(.rsaPublicKeys).Run()
	if  != nil {
		return 
	}

	.sessionMux.Lock()
	.authKey = .AuthKey
	.sessionID = .SessionID
	.salt = .ServerSalt
	.sessionMux.Unlock()

	return nil
}
