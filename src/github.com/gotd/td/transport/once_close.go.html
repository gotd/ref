// onceCloseListener wraps a net.Listener, protecting it from
// multiple Close calls.
type onceCloseListener struct {
	net.Listener
	once sync.Once
	err  error
}

func ( *onceCloseListener) () error {
	.once.Do(func() {
		.err = .Listener.Close()
	})
	return .err
}
