// GIF add attributes to create GIF attachment.
func ( *UploadedDocumentBuilder) () *UploadedDocumentBuilder {
	return .Attributes(&tg.DocumentAttributeAnimated{}).
		MIME(DefaultGifMIME)
}

// GIF adds gif attachment.
func ( tg.InputFileClass,  ...StyledTextOption) *UploadedDocumentBuilder {
	return UploadedDocument(, ...).GIF()
}

// GIF sends gif.
func ( *Builder) (
	 context.Context,
	 tg.InputFileClass,  ...StyledTextOption,
) (tg.UpdatesClass, error) {
	return .Media(, GIF(, ...))
}
