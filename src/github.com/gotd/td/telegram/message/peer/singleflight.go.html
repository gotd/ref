type singleFlight struct {
	next Resolver
	sg   singleflight.Group
}

func ( *singleFlight) ( context.Context,  string) (tg.InputPeerClass, error) {
	 := .sg.DoChan(, func() (interface{}, error) {
		return .next.ResolveDomain(, )
	})

	select {
	case <-.Done():
		return nil, .Err()
	case  := <-:
		if .Err != nil {
			return nil, .Err
		}
		return .Val.(tg.InputPeerClass), nil
	}
}
