type testTransportHandler struct {
	t      testing.TB
	logger *zap.Logger
	// For ACK testing proposes.
	// We send ack only after second request
	counter   int
	counterMx sync.Mutex

	message string // immutable
}

func ( testing.TB,  *zap.Logger,  string) Handler {
	return &testTransportHandler{
		t:       ,
		logger:  ,
		message: ,
	}
}

func ( *testTransportHandler) ( *Server,  *Request) error {
	,  := .Buf.PeekID()
	if  != nil {
		return 
	}

	.logger.Info("New message", zap.String("id", fmt.Sprintf("%x", )))

	switch  {
	case tg.InvokeWithLayerRequestTypeID:
		 := tg.InvokeWithLayerRequest{
			Query: &tg.InitConnectionRequest{
				Query: &tg.HelpGetConfigRequest{},
			},
		}

		if  := .Decode(.Buf);  != nil {
			return 
		}
		.logger.Info("New client connected, invoke received")

		if  := .SendConfig();  != nil {
			return 
		}

		.logger.Info("Sending message", zap.String("message", .message))
		return .SendUpdates(.RequestCtx, .Session, &tg.UpdateNewMessage{
			Message: &tg.Message{
				ID:      1,
				PeerID:  &tg.PeerUser{UserID: 1},
				Message: .message,
			},
		})
	case tg.MessagesSendMessageRequestTypeID:
		 := &tg.MessagesSendMessageRequest{}
		if  := .Decode(.Buf);  != nil {
			.t.Fail()
			return 
		}

		require.Equal(.t, "какими деньгами?", .Message)

		.counterMx.Lock()
		.counter++
		if .counter < 2 {
			.counterMx.Unlock()
			return nil
		}
		.counterMx.Unlock()

		return .SendResult(, &tg.Updates{})
	}

	return nil
}
