// ZapPreferNoWith suggests replace expressions like
//
// 	l.With(...).Debug("")
//
// to
//
// 	l.Debug("", ...).
//
// where l is a *zap.Logger.
func ( dsl.Matcher) {
	.Import("go.uber.org/zap")

	.Match("$l.With($*args).$method($*msg_args)").Where(
		["l"].Type.Is("*zap.Logger") &&
			["method"].Text.Matches("Debug|Info|Warn|Error|DPanic|Panic|Fatal"),
	).Suggest("$l.$method($msg_args, $args)")
}
